\begin{minipage}[t]{0.323\textwidth}\vspace{0pt}
\begin{poscol}{1\quad Motivation \& background}
\begin{minipage}[t][\colheight][s]{\linewidth}
Two numbers govern how a structural alloy carries load --- and a neural
network can predict both from composition alone.

\smallskip
\begin{center}
\resizebox{0.5\linewidth}{!}{%
\begin{tikzpicture}[line cap=round, line join=round, scale=1.4,
  pull/.style={-{Stealth[length=2.8mm]}, accent, line width=1.5pt},
  dim/.style={{Stealth[length=1.8mm]}-{Stealth[length=1.8mm]}, ink!75,
    line width=0.7pt}]
  \fill[ink!7] (0,0) rectangle (1.3,2.0);
  \draw[ink, line width=1pt] (0,0) rectangle (1.3,2.0);
  \node[font=\footnotesize, ink!70] at (0.65,-0.4) {unloaded};
  \draw[-{Stealth[length=3.2mm]}, accent!75, line width=1.8pt]
        (1.7,1.0) -- (2.7,1.0);
  \draw[ink!40, dashed, line width=0.8pt] (3.25,0) rectangle (4.55,2.0);
  \fill[accent!9] (3.45,-0.35) rectangle (4.35,2.55);
  \draw[accent, line width=1.5pt] (3.45,-0.35) rectangle (4.35,2.55);
  \draw[pull] (3.9,2.55) -- (3.9,3.5) node[above, font=\footnotesize]{$\sigma$};
  \draw[pull] (3.9,-0.35) -- (3.9,-1.05) node[below, font=\footnotesize]{$\sigma$};
  \draw[dim] (4.8,0) -- (4.8,2.55);
  \node[font=\footnotesize, ink!75, right] at (4.85,1.3) {$\varepsilon_\parallel$};
  \draw[dim] (3.45,2.82) -- (4.35,2.82);
  \node[font=\footnotesize, ink!75, left] at (3.38,2.82) {$\varepsilon_\perp$};
\end{tikzpicture}%
}
\end{center}
\vspace{0.1em}

\begin{tight}
  \item \textbf{Young's modulus} $E=\sigma/\varepsilon_\parallel$ --- stiffness,
        the stress needed to stretch it (GPa).
  \item \textbf{Poisson's ratio} $\nu=-\varepsilon_\perp/\varepsilon_\parallel$
        --- how much it thins sideways when pulled.
\end{tight}

\smallskip
For a cubic crystal both reduce to two stiffness constants ---
$C_{11}$ (axial) and $C_{12}$ (transverse):
\begin{gather*}
E=\frac{(C_{11}-C_{12})(C_{11}+2C_{12})}{C_{11}+C_{12}},\\[2pt]
\nu=\frac{C_{12}}{C_{11}+C_{12}},\qquad B=\tfrac{1}{3}(C_{11}+2C_{12}).
\end{gather*}
\begin{tight}
  \item Stiff alloy ($E$ large) resists stretching; low $\nu$ barely thins
        under load.
  \item $E$ and $\nu$ fix deflection, vibration and buckling --- the first
        properties screened for a structural alloy, and the targets this
        pipeline predicts and optimizes.
\end{tight}

\vfill
\subhead{The bottleneck}
\begin{tight}
  \item Molecular dynamics recovers $E$ and $\nu$ cheaply --- but only with an
        accurate interatomic potential.
  \item The standard metallic form, the Modified Embedded Atom Method
        (MEAM) [1,\,2], is fitted \emph{by hand}.
  \item Hand fitting is expert work that does not scale past a few elements.
\end{tight}

\begin{keyresult}
\textbf{Objective.} Build MEAM potentials \emph{automatically} for any
set of elements --- no hand fitting, no experimental input.
\end{keyresult}

\vfill
\subhead{The design space}
\begin{tight}
  \item Compositions drawn on the 12-element simplex (Al, Co, Cr, Cu, Fe, Mg,
        Mn, Mo, Ni, Si, Ti, Zn), weighted toward aluminium.
  \item Of ${\sim}7\,900$ random configurations, \textbf{7\,815 survive} the
        physicality filter ($E>0$, $0<\nu<0.48$, $C_{11}>C_{12}$).
\end{tight}

\begin{minipage}[t]{0.323\textwidth}\vspace{0pt}
\begin{poscol}{2\quad Methodology}
\begin{minipage}[t][\colheight][s]{\linewidth}
An automated five-stage pipeline turns random compositions into an
optimized multi-element potential.

\begin{methodsbox}
{\color{accent}\bfseries Three engines at a glance}
\begin{tight}
  \item \textbf{DFT} (Quantum Espresso) --- first-principles $a_0$,
        $E_{\mathrm{coh}}$, $B$ and binary formation energies: the
        ground-truth references.
  \item \textbf{MEAM} --- many-body metallic potential (Rose binding curve
        $+$ cross-terms), DFT-seeded and neural-network refit.
  \item \textbf{MD} (LAMMPS) --- strain sweep on each potential
        $\to C_{11},C_{12}\to E,\nu$: the elastic database.
\end{tight}
\end{methodsbox}

\smallskip
\begin{center}
\resizebox{0.82\linewidth}{!}{%
\begin{tikzpicture}[

\vfill
\subhead{DFT-seeded initialisation}
\begin{tight}
  \item Quantum Espresso [4] fixes each element's $a_0$, $E_{\mathrm{coh}}$ and
        bulk modulus $B$, pinning the Rose binding curve [5] and the MEAM
        decay exponent
        \[ \alpha=\sqrt{\,9B\Omega/E_{\mathrm{coh}}\,} \]
        ($\Omega$ the atomic volume; $\alpha\!\approx\!4.5$ for Al, up to
        $10.9$ for Zn).
  \item Sixty-six binary formation energies seed each cross-term's sign.
\end{tight}

\begin{center}
\includegraphics[width=0.98\linewidth]{figures/formation_energy_heatmap.png}
\figcap{DFT binary formation energies (eV/atom); negative $=$ exothermic.}
\end{center}

\vfill
\subhead{Neural surrogate \& inverse design}
\begin{tight}
  \item A fully connected network maps composition to the stiffness constants
        ($E$, $\nu$ follow analytically).
  \item A second network learns (MEAM parameters)$\,\to(C_{11},C_{12})$.
  \item Gradient descent through the \emph{frozen} surrogate then locates
        parameters that hit a target.
\end{tight}

\begin{minipage}[t]{0.323\textwidth}\vspace{0pt}
\begin{poscol}{3\quad Results \& discussion}
\begin{minipage}[t][\colheight][s]{\linewidth}
\subhead{Elastic database}
\begin{tight}
  \item ${\sim}4$\,nm LAMMPS [3] supercell per alloy; axial strains
        ($\delta=10^{-3}$) give $C_{11},C_{12}$ and hence $E$ and
        $\nu=C_{12}/(C_{11}+C_{12})$.
  \item Median $E=103.6$\,GPa, $\nu=0.326$ over the corpus.
\end{tight}

\begin{center}
\includegraphics[width=\linewidth]{figures/E_nu_scatter.png}
\figcap{$E$--$\nu$ for 7\,815 alloys, coloured by dominant element;
commercial Al alloys starred.}
\end{center}

\vfill
\subhead{Composition surrogate}
\begin{keyresult}
$R^{2}_{E}=0.77$, $R^{2}_{\nu}=0.52$ on held-out validation. For
Al\,7075 the surrogate gives $E=75.5$\,GPa, $\nu=0.334$ vs published
$71$--$72$\,GPa, $\nu\approx0.33$ --- within ${\sim}5\%$.
\end{keyresult}

\vfill
\subhead{Optimized potential}
\begin{center}
\includegraphics[width=\linewidth]{figures/nnip_parity.png}
\figcap{Optimized potential vs MD on 60 unseen alloys: inside the
$\pm20\%$ band $=$ match, $\times$ $=$ collapsed (unphysical).}
\end{center}
\begin{keyresult}
\textbf{Proof of concept.} Of \textbf{60} unseen alloys, \textbf{25}
stayed physically stable; \textbf{17} matched $\nu$ and \textbf{8}
matched $E$ within ${\sim}20\%$.
\end{keyresult}

\vfill
\subhead{Conclusions \& outlook}
\begin{itemize}\setlength\itemsep{0.25em}\setlength\leftmargini{1.1em}
  \item \textbf{Automated end-to-end 12-element MEAM pipeline} --- random
        compositions to an optimized potential, no hand fitting.
  \item \textbf{Al-alloy elastic prediction} from composition alone,
        within a few percent of published values.
  \item \textbf{DFT-seeded, NN-trained MEAM} --- first-principles
        references refit through a neural surrogate.
  \item Next: more data, longer training, denser DFT references.
\end{itemize}
