\begin{frame}{Young's Modulus $E$ and Poisson's Ratio $\nu$}
\centering\small
A material's elastic response is captured by two numbers. For a cubic
crystal both follow from the stiffness constants $C_{11}$ and $C_{12}$,
the quantities this work predicts.

\medskip
\begin{columns}[c]

% ---------------- Young's modulus ----------------
\column{0.5\textwidth}
\centering
\textbf{$E$: resistance to stretching}

\smallskip
\begin{tikzpicture}[>=Stealth, line width=0.8pt, font=\scriptsize]
  \useasboundingbox (-0.4,-1.25) rectangle (3.9,1.25);
  % fixed wall with hatching
  \draw[gray!70] (0,-0.55) -- (0,0.55);
  \foreach \y in {-0.5,-0.3,...,0.55}{\draw[gray!55] (0,\y) -- (-0.18,\y-0.18);}
  % stretched bar
  \draw[darkred, fill=darkred!8] (0,-0.3) rectangle (2.7,0.3);
  % original right edge (before load)
  \draw[gray!65, dashed] (2.0,-0.5) -- (2.0,0.5);
  % applied force
  \draw[->, very thick, darkred] (2.7,0) -- (3.6,0) node[right]{$F$};
  % dimensions
  \draw[<->, gray!70] (0,-0.68) -- (2.0,-0.68) node[midway,below]{$L$};
  \draw[<->, darkred] (2.0,0.62) -- (2.7,0.62) node[midway,above]{$\Delta L$};
\end{tikzpicture}

\smallskip
$E = \dfrac{\sigma}{\varepsilon} = \dfrac{F/A}{\Delta L/L}$

\smallskip
{\scriptsize Large $E$ means stiff: steel $\approx 200$, Al $\approx 70$\,GPa.}

% ---------------- Poisson's ratio ----------------
\column{0.5\textwidth}
\centering
\textbf{$\nu$: sideways thinning when pulled}

\smallskip
\begin{tikzpicture}[>=Stealth, line width=0.8pt, font=\scriptsize]
  \useasboundingbox (-1.55,-1.25) rectangle (1.55,1.25);
  % original shape (before load)
  \draw[gray!60, dashed] (-0.62,-0.55) rectangle (0.62,0.55);
  % loaded shape: taller and narrower
  \draw[deepblue, fill=deepblue!8] (-0.40,-0.82) rectangle (0.40,0.82);
  % axial pull
  \draw[->, very thick, deepblue] (0,0.82) -- (0,1.18) node[above]{$F$};
  \draw[->, very thick, deepblue] (0,-0.82) -- (0,-1.18) node[below]{$F$};
  % lateral contraction (dashed edge -> loaded edge)
  \draw[->, darkred] (-0.62,0) -- (-0.42,0);
  \draw[->, darkred] (0.62,0) -- (0.42,0);
  \node[deepblue, font=\tiny, align=center, right] at (0.66,0.45){axial\\stretch};
  \node[darkred, font=\tiny, align=center, left] at (-0.66,-0.45){lateral\\contraction};
\end{tikzpicture}

\smallskip
$\nu = -\dfrac{\varepsilon_{\text{lateral}}}{\varepsilon_{\text{axial}}}$

\smallskip
{\scriptsize Stretch it and it thins: most metals $\nu \approx 0.3$.}

\end{columns}

\vspace{0.3em}
\centering{\footnotesize Mechanical stability of a cubic crystal requires
$C_{11} > C_{12}$ (Born--Cauchy).}
\end{frame}
