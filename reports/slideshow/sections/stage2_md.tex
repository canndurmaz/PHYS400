\begin{frame}{Stage 2: MD Simulation \normalfont\small\textcolor{gray}{--- LAMMPS (Python API) + OVITO}}
\textbf{Static deformation method} to compute elastic properties:

\begin{enumerate}
  \item \textbf{Relax} --- anisotropic pressure minimization (box + atoms)
  \item \textbf{Baseline} --- record stress $\sigma_{xx}, \sigma_{yy}, \sigma_{zz}$
  \item \textbf{Strain} --- apply $\pm\delta$ ($\delta = 0.001$) along $x$, $y$, $z$ with re-minimization
  \item \textbf{Extract} --- $C_{11}$ from axial, $C_{12}$ from transverse stress derivatives
  \item \textbf{Average} --- 3 $C_{11}$ and 6 $C_{12}$ samples across directions
  \item \textbf{Compute:}
\end{enumerate}

\vspace{-0.3em}
\begin{equation*}
  E = \frac{(C_{11} - C_{12})(C_{11} + 2C_{12})}{C_{11} + C_{12}}, \qquad
  \nu = \frac{C_{12}}{C_{11} + C_{12}}
\end{equation*}

\vspace{0.3em}
Configs with $\nu < 0$ are discarded as unphysical.
\end{frame}
