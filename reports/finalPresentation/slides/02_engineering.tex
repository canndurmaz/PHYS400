\begin{frame}{Atomistic Modelling: Interatomic Potentials and DFT}
\begin{columns}[T]
\column{0.49\textwidth}
\textbf{Interatomic potential.}
Closed-form approximation of the many-body potential energy
$U(\{\mathbf{r}_{i}\})$;
forces obtained as $\mathbf{F}_{i} = -\nabla_{i}U$.

\smallskip
\textbf{Historical development:}
\begin{itemize}\scriptsize
  \item Pairwise (Lennard--Jones, Morse) ---
        inadequate for metallic bonding.
  \item Embedded Atom Method (Daw \& Baskes, 1984) ---
        suitable for FCC metals.
  \item \textbf{Modified EAM}~\citep{baskes1992,lee2001}: introduces
        angular partial densities $\rho_{i}^{(l)}$, $l = 1{-}3$;
        established formalism for BCC, HCP and mixed-bonding
        metals.
\end{itemize}

\column{0.49\textwidth}
\textbf{Density-functional theory (DFT).}
First-principles electronic-structure method, formulated via the
Kohn--Sham equations~\citep{kohn1965} and implemented here with a
plane-wave + pseudopotential basis (QE,
\citealt{giannozzi2009}).

\smallskip
Quantities computed:
\begin{itemize}\scriptsize
  \item Cohesive energy $E_{\mathrm{coh}}$
  \item Equilibrium lattice $a_{0}$, bulk modulus $B$
  \item Binary formation energies $E_{\mathrm{form}}(i, j)$
  \item Single-crystal elastic constants $C_{11}, C_{12}$
\end{itemize}

\smallskip
\begin{block}{}
\scriptsize\centering
DFT: high accuracy, limited scale ($\le 10^{3}$ atoms).\\
MEAM/MD: empirical, large scale ($\ge 10^{6}$ atoms).\\[0.2em]
\emph{DFT provides the reference dataset against which the MEAM
potential is calibrated.}
\end{block}
\end{columns}
\end{frame}
