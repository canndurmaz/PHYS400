% --- Shared roadmap layout (agenda slide + section dividers) ----------
% Two zones on a [plain] frame: a darkred accent panel on the left
% (#1 = its content) and a section list on the right (#2). The agenda and
% the dividers call this with different panel content and TOC variants, so
% the two slide types share one visual language and cannot drift apart.
\newcommand{\rmframe}[2]{%  #1 = left-panel content, #2 = toc invocation
  \begin{frame}[plain]
    \begin{tikzpicture}[remember picture, overlay]
      \fill[darkred!7]
        (current page.north west) rectangle
        ([xshift=0.36\paperwidth]current page.south west);
      \draw[darkred, line width=1.3pt]
        ([xshift=0.36\paperwidth]current page.north west) --
        ([xshift=0.36\paperwidth]current page.south west);
      \node[anchor=west, text width=0.27\paperwidth, align=left]
        at ([xshift=0.05\paperwidth]current page.west) {#1};
      % METU logo (plain frames skip the frametitle template that adds it)
      \node[anchor=north east, xshift=-10pt, yshift=-9pt]
        at (current page.north east) {\includegraphics[height=0.7cm]{metu.png}};
    \end{tikzpicture}%
    \hspace*{0.42\paperwidth}%
    % Bottom strut lifts the centred list slightly above the page centre so
    % the last item does not graze the lower edge.
    \begin{minipage}[c][\paperheight][c]{0.52\paperwidth}%
      #2\par\vspace*{0.13\paperheight}%
    \end{minipage}%
  \end{frame}}

% --- Recurring "where are we now" section divider ---
% Fires automatically at every \section. The \ifnum guard skips the first
% section (Motivation), already covered by the standalone agenda slide, so
% dividers appear only when transitioning into a new section. Left panel:
% large 0-padded section number + section title. Right: current-highlighted
% section list.
\AtBeginSection[]{%
  \ifnum\value{section}>1
    \rmframe{%
      {\fontsize{64}{64}\selectfont\bfseries\color{darkred!85}0\arabic{section}}\\[0.45em]
      {\small\bfseries\color{darkred!55}\insertsection}%
    }{\tableofcontents[currentsection]}%
  \fi
}

% --- Per-stage framework tag -----------------------------------------
% A small darkred pill seated in the empty right region of the title bar
% (left of the METU logo) naming the software framework that powers a
% stage. Single line so it clears the title bar without touching content.
% Call \fwtag{<name>} right after \begin{frame}{...} on each stage slide.
\newcommand{\fwtag}[1]{%
  \begin{tikzpicture}[remember picture, overlay]
    \node[anchor=east, fill=darkred, text=white, rounded corners=2pt,
          inner xsep=6pt, inner ysep=2.5pt, font=\scriptsize\bfseries]
      at ([xshift=-1.15cm, yshift=-0.40cm]current page.north east) {#1};
  \end{tikzpicture}%
}
