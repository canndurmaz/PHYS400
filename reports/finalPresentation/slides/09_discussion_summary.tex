\begin{frame}{Discussion and Outlook}
\begin{columns}[T]
\column{0.32\textwidth}
\textbf{Contributions}
\begin{itemize}\scriptsize
  \item MD elastic database, 7815 alloys
  \item DFT reference set, 10 elements
  \item Merged 12-element MEAM library
  \item Composition surrogate
        ($R^{2}_{E} = 0.77$, deployed)
  \item Resumable Stage~5b framework
\end{itemize}

\column{0.32\textwidth}
\textbf{Current limitations}
\begin{itemize}\scriptsize
  \item 4/30 predictions within tolerance
  \item 19/30 violate Born--Cauchy
  \item Cross-term parameters saturate bounds
\end{itemize}

\smallskip
\textbf{Identified causes}
\begin{itemize}\scriptsize
  \item Under-determined inverse problem
        ($\sim\!200$ parameters, $\le\!100$ targets)
  \item Surrogate--LAMMPS residual mismatch
  \item Stability constraints absent from the loss
\end{itemize}

\column{0.34\textwidth}
\textbf{Planned extensions}
\begin{itemize}\scriptsize
  \item Restrict optimisation to single-element-anchored
        cross-terms ranked by Sobol sensitivity
  \item Active-learning expansion to $\ge\!500$ medoids
  \item Physics-informed loss with stability penalties
  \item SQS DFT references~\citep{lewis1995}
        at intermediate compositions
\end{itemize}
\end{columns}
\end{frame}
