\begin{frame}{Stage 5b: NN-Driven MEAM Parameter Optimisation}
\fwtag{TensorFlow + LAMMPS}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Closed-loop procedure}
\begin{enumerate}\small
  \item \textbf{Surrogate.} A small MLP learns
        $(\text{MEAM params}) \to (C_{11}, C_{12})$.
  \item \textbf{Fit.} Train it with Adam~\citep{kingma2015adam} on a
        Huber loss.
  \item \textbf{Invert.} Descend through the \emph{frozen} surrogate to
        the parameters that hit the target $(C_{11}, C_{12})$.
  \item \textbf{Loop.} Re-evaluate in LAMMPS, append samples, retrain.
\end{enumerate}

\medskip
{\footnotesize Adam adapts the step size per parameter; the Huber loss
keeps outlier (Mg/Zn) configurations from dominating the fit.}

\column{0.5\textwidth}
\centering
\includegraphics[width=0.80\linewidth]{figures/nnip_training_loss.png}\\[-0.3em]
\scriptsize Surrogate Huber loss decays $0.30 \to 0.16$ over training.

\smallskip
\begin{block}{Result: a working proof of concept}
\small
Tested on \textbf{30 unseen alloys}: \textbf{11} were physically stable,
and \textbf{4} matched the MD reference within $\sim\!20\,\%$. Accuracy
grows with more data and longer training.
\end{block}
\end{columns}

\note{Surrogate: 4-layer MLP (20-20-10, ReLU). Fit with Adam on a Huber
loss (delta = 1), up to 1e4 epochs, early-stopped below 5e-3. Inverse
step: Adam, lr 0.01, 3000 steps, descending through the frozen surrogate.
Active learning: optimised parameters are re-evaluated in LAMMPS and
appended to the training set each round.}
\end{frame}
