\begin{frame}{Further Studies}
\begin{columns}[T]
\column{0.32\textwidth}
\textbf{Well-posed optimisation}
\begin{itemize}\scriptsize
  \item Restrict to single-element-anchored
        cross-terms, ranked by Sobol sensitivity
        ($\sim\!200$ params $\to$ a tractable few)
  \item Physics-informed loss: Born--Cauchy
        stability penalties baked into training
  \item Active-learning expansion to
        $\ge\!500$ composition medoids
\end{itemize}

\column{0.34\textwidth}
\textbf{Beyond the aluminium domain}
\begin{itemize}\scriptsize
  \item Extend the basis to out-of-basis
        elements (C, V, N) so ferrous and
        Ti-alloy chemistries are representable
  \item Broaden MD/DFT sampling into
        \emph{hcp} (Ti, Mg, Zn) and ferrous
        (Fe--Cr--Ni) regions
  \item Symmetry-aware elastic extraction
        (hexagonal tensor) --- removes the
        cubic-strain error on hcp metals
\end{itemize}

\column{0.32\textwidth}
\textbf{Reference data \& validation}
\begin{itemize}\scriptsize
  \item SQS DFT references~\citep{lewis1995}
        at intermediate compositions
  \item Enlarged DFT set spanning the
        broadened element domain
  \item Re-validate each round against
        ASM/MatWeb experiment --- a standing
        external accuracy loop
\end{itemize}
\end{columns}

\vspace{0.4em}
{\footnotesize The architecture is established; the remaining work is
\emph{sampling}, \emph{loss formulation}, and \emph{domain coverage}.}
\end{frame}
