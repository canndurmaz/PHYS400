\begin{frame}{Stage 3: The Quantum Espresso Framework}
\textbf{Quantum Espresso}~\citep{giannozzi2009}\footnote{\tiny QE = Quantum Espresso --- open-source plane-wave DFT.}
solves Kohn--Sham DFT~\citep{kohn1965} in a \emph{plane-wave +
pseudopotential} basis; QE\,7.5 is driven through the
\textbf{Atomic Simulation Environment} (ASE)~\citep{ase2017}
Python interface, so single-shot SCFs, EOS scans and elastic
calculations are dispatched from a single process pool.

\medskip
\begin{columns}[T]
\column{0.49\textwidth}
\textbf{Computational settings}
\begin{itemize}\small
  \item Plane-wave cut-off: $E_{\mathrm{cut}} = 40$\,Ry
        (charge $= 320$\,Ry)
  \item Marzari--Vanderbilt smearing~\citep{marzari1999}, width $0.02$\,Ry
  \item Monkhorst--Pack mesh~\citep{monkhorst1976}: $6\!\times\!6\!\times\!6$
  \item Spin-polarised auto-on for Fe, Cr, Mn
  \item Pseudopotentials drawn from the official QE library on demand
\end{itemize}

\column{0.49\textwidth}
\textbf{Per-element computation}
\begin{enumerate}\small
  \item Seven-point EOS scan
        $\to$ $a_0, E_{\mathrm{coh}}, B$ via
        \textbf{Birch--Murnaghan} fit~\citep{birch1947,birch}.
  \item Isolated-atom reference $\to$ true $E_{\mathrm{coh}}$.
  \item Strain protocol $\to$ $C_{11}, C_{12}$ (same as LAMMPS).
\end{enumerate}
\textbf{Per binary pair.} A single SCF on the
$L1_2$/$B2$ reference structure $\to$ formation energy
$E_{\mathrm{form}}(i,j)$.

\medskip
\textbf{Validation.} QE--ASE coupling sanity-checked on bulk FCC
Al: $E \in [-540,-530]$\,eV, $|F| < 0.01$\,eV/\AA,
diagonal-only stress. All pass.
\end{columns}
\end{frame}

$E_{\mathrm{form}}(i,j)$ in eV/atom from $L1_2$/$B2$ SCF

\medskip
\textbf{Negative (blue)} $\Rightarrow$ exothermic mixing
(e.g.\ Al--Ti, Cu--Ti --- compatible $sd$/$sp$ pairs).

\medskip
\textbf{Strongly positive (red)} $\Rightarrow$ poor mutual
solubility (notably Mg and Zn against most transition metals;
Fe--Mo also stands out).

\medskip
These signed values seed the cross-term $E_c(i,j)$ entries of
the merged MEAM library directly, with no further fitting.
